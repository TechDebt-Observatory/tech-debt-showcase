/*
 * ============================================================================
 * ASN1PARSE - OpenSSL ASN.1 Structure Parser and Diagnostic Tool
 * ============================================================================
 * 
 * PURPOSE:
 *   This utility parses and displays the structure of ASN.1 encoded data.
 *   ASN.1 (Abstract Syntax Notation One) is the encoding used in X.509 
 *   certificates, PKCS structures, and many other cryptographic formats.
 *
 * PRIMARY USE CASES:
 *   - Debugging certificate and key file formats
 *   - Examining ASN.1 structure of encoded data
 *   - Generating ASN.1 structures from configuration files
 *   - Converting between different encodings (PEM, DER, Base64)
 *   - Drilling down into nested ASN.1 structures
 *
 * ============================================================================
 * INPUTS AND OUTPUTS
 * ============================================================================
 *
 * COMMAND LINE INPUTS:
 *   -in          Input file containing ASN.1 data (stdin if not specified)
 *   -inform    Input format: PEM (default), DER, or B64 (Base64)
 *   -out         Output file for DER data (optional)
 *   -offset         Starting byte offset into input file
 *   -length         Number of bytes to process (0 = all remaining)
 *   -strparse       Offset to parse nested structure (can be repeated)
 *   -oid         File containing additional OID definitions
 *   -genstr    Generate ASN.1 from string specification
 *   -genconf     Generate ASN.1 from configuration file
 *   -item        Parse as specific ASN.1 type (e.g., X509, RSAPublicKey)
 *   -noout             Suppress output display
 *   -i                 Indent the output for readability
 *   -dump              Display unknown data in hex format
 *   -dlimit         Limit hex dump to first N bytes
 *
 * FILE INPUTS:
 *   - ASN.1 encoded data (PEM, DER, or Base64 format)
 *   - Optional OID definition file (text format)
 *   - Optional configuration file for ASN.1 generation
 *
 * OUTPUTS:
 *   STDOUT: Human-readable ASN.1 structure breakdown showing:
 *           - Tag numbers and types (SEQUENCE, INTEGER, etc.)
 *           - Lengths of each element
 *           - Hierarchical structure with indentation
 *           - Hex dump of primitive data (if -dump specified)
 *           - Offset positions within the file
 *
 *   -out file: Raw DER encoded data (if specified)
 *              Useful for extracting portions of larger structures
 *
 *   STDERR: Error messages and OpenSSL error stack
 *
 * RETURN VALUES:
 *   0 = Success
 *   1 = Error (invalid input, parse failure, file I/O error, etc.)
 *
 * EXAMPLES:
 *   # Parse a PEM certificate
 *   openssl asn1parse -in cert.pem
 *
 *   # Parse DER format with hex dump
 *   openssl asn1parse -inform DER -in data.der -dump
 *
 *   # Extract and parse nested structure at offset 120
 *   openssl asn1parse -in cert.pem -strparse 120
 *
 *   # Drill down through multiple levels
 *   openssl asn1parse -in cert.pem -strparse 4 -strparse 20
 *
 *   # Generate ASN.1 from specification
 *   openssl asn1parse -genstr "SEQUENCE:seq" -out generated.der
 *
 * ============================================================================
 * VARIABLE DICTIONARY - asn1parse_main()
 * ============================================================================
 *
 * COMMAND LINE & CONFIGURATION:
 *   argc             [in]  Command line argument count
 *   argv             [in]  Command line argument array
 *   prog             [local] Program name for error messages
 *   o                [local] Current option being processed (OPTION_CHOICE enum)
 *   infile           [local] Input file path (NULL = stdin)
 *   oidfile          [local] OID definitions file path
 *   derfile          [local] Output file path for DER data
 *   genstr           [local] ASN.1 generation string from command line
 *   genconf          [local] Config file path for ASN.1 generation
 *
 * CONTROL FLAGS:
 *   informat         [local] Input format: FORMAT_PEM, FORMAT_ASN1, FORMAT_BASE64
 *   indent           [local] Boolean: indent output (1) or not (0)
 *   noout            [local] Boolean: suppress output display
 *   dump             [local] Hex dump control: 0=none, -1=unlimited, >0=limit bytes
 *   ret              [local] Return code: 0=success, 1=error
 *
 * DATA BUFFER MANAGEMENT:
 *   buf              [local] Main memory buffer (BUF_MEM*) for holding all data
 *   str              [local] Pointer to actual data bytes (unsigned char*)
 *   num              [local] Total number of bytes of data (long)
 *   offset           [local] Starting offset into data for parsing
 *   length           [local] Number of bytes to parse (0 = all remaining)
 *
 * BIO (I/O) STREAMS:
 *   in               [local] Input BIO stream (file or stdin)
 *   b64              [local] Base64 decoding filter BIO (if needed)
 *   derout           [local] Output BIO stream for DER data (if -out specified)
 *
 * PEM FORMAT PARSING:
 *   name             [local] PEM label (e.g., "CERTIFICATE", "PRIVATE KEY")
 *   header           [local] PEM header lines (metadata)
 *
 * ASN.1 PARSING:
 *   at               [local] Current ASN.1 structure being parsed (ASN1_TYPE*)
 *   it               [local] ASN.1 item type for typed parsing (ASN1_ITEM*)
 *   ctmpbuf          [local] Const pointer for d2i parsing functions
 *
 * STRPARSE (NESTED STRUCTURE DRILLING):
 *   osk              [local] Stack of strparse offset strings (STACK_OF(OPENSSL_STRING)*)
 *   tmpbuf           [local] Temporary pointer during strparse iteration
 *   tmplen           [local] Temporary length during strparse iteration
 *
 * LOOP VARIABLES:
 *   i                [local] General purpose loop counter / bytes read
 *   j                [local] Parsed strparse offset value
 *
 * ============================================================================
 * VARIABLE DICTIONARY - do_generate()
 * ============================================================================
 *
 * FUNCTION PARAMETERS:
 *   genstr           [in]  ASN.1 generation string (e.g., "SEQUENCE:section")
 *   genconf          [in]  Configuration file path (or NULL)
 *   buf              [out] Buffer to receive generated DER data
 *
 * LOCAL VARIABLES:
 *   cnf              [local] Configuration file object (CONF*)
 *   atyp             [local] Generated ASN.1 structure (ASN1_TYPE*)
 *   len              [local] Length of DER encoded output
 *   p                [local] Pointer for DER encoding operation
 *
 * RETURN VALUE:
 *   Positive integer: Length of generated DER data (success)
 *   -1: Error occurred
 *
 * ============================================================================
 * MEMORY MANAGEMENT NOTES:
 * ============================================================================
 *
 * ALLOCATED RESOURCES (must be freed at 'end' label):
 *   - buf: BUF_MEM buffer (freed by BUF_MEM_free)
 *   - in, b64, derout: BIO streams (freed by BIO_free)
 *   - name, header: PEM strings (freed by OPENSSL_free)
 *   - at: ASN1_TYPE structure (freed by ASN1_TYPE_free)
 *   - osk: String stack (freed by sk_OPENSSL_STRING_free)
 *
 * SPECIAL CASES:
 *   - str: May point into buf->data (PEM path) or be separately allocated
 *          Only freed via BUF_MEM_free(buf)
 *   - tmpbuf: Points into 'at' structure's internal data during strparse
 *             Not separately freed (lifetime tied to 'at')
 *
 * ERROR HANDLING:
 *   All error paths use 'goto end' to ensure proper cleanup
 *   This is a classic C pattern for resource management
 *
 * ============================================================================
 * TECHNICAL DEBT MARKERS:
 * ============================================================================
 *
 * Throughout this code, you'll see comments marked "TECHNICAL DEBT"
 * These indicate known issues that should be addressed:
 *
 * 1. Unchecked strtol() calls - No error handling for integer conversions
 * 2. Integer overflow checks after operations - Should be before
 * 3. Uninitialized variables - Some loop counters not initialized
 * 4. Type mixing - Signed/unsigned integer comparisons
 * 5. Magic numbers - Unexplained constants like "BUFSIZ * 8"
 * 6. "Evil" pointer manipulation - Direct access to internal structures
 * 7. Goto-heavy control flow - Makes code harder to follow
 *
 * See full analysis at: [your-project-url]
 *
 * ============================================================================
 */

static int do_generate(char *genstr, const char *genconf, BUF_MEM *buf)
{
    CONF *cnf = NULL;           /* Configuration object */
    int len;                    /* Length of generated data */
    unsigned char *p;           /* Pointer for encoding */
    ASN1_TYPE *atyp = NULL;     /* Generated ASN.1 structure */

    if (genconf != NULL) {
        /* Load configuration file */
        if ((cnf = app_load_config(genconf)) == NULL)
            goto err;
        
        /* If no explicit genstr, read from config file */
        if (genstr == NULL)
            genstr = NCONF_get_string(cnf, "default", "asn1");
        
        if (genstr == NULL) {
            BIO_printf(bio_err, "Can't find 'asn1' in '%s'\n", genconf);
            goto err;
        }
    }

    /* Generate ASN.1 structure from string specification */
    atyp = ASN1_generate_nconf(genstr, cnf);
    NCONF_free(cnf);
    cnf = NULL;

    if (atyp == NULL)
        return -1;

    /*
     * Encode to DER format
     * First pass with NULL pointer returns required length
     */
    len = i2d_ASN1_TYPE(atyp, NULL);

    if (len <= 0)
        goto err;

    /* Grow buffer to hold encoded data */
    if (!BUF_MEM_grow(buf, len))
        goto err;

    /* Second pass actually encodes the data */
    p = (unsigned char *)buf->data;
    i2d_ASN1_TYPE(atyp, &p);

    ASN1_TYPE_free(atyp);
    return len;

 err:
    /* Cleanup on error */
    NCONF_free(cnf);
    ASN1_TYPE_free(atyp);
    return -1;

/*
 * asn1parse_main - Main entry point for the asn1parse command
 *
 * This function orchestrates the parsing and display of ASN.1 encoded data.
 * It handles multiple input formats (PEM, DER, Base64), can generate ASN.1
 * structures from configuration, and supports "drilling down" into nested
 * ASN.1 structures using the strparse option.
 *
 * Arguments:
 *   argc - Command line argument count
 *   argv - Command line argument vector
 *
 * Returns:
 *   0 on success
 *   1 on error
 */
int asn1parse_main(int argc, char **argv)
{
    /* ASN.1 type structure - holds parsed ASN.1 data */
    ASN1_TYPE *at = NULL;
    
    /* BIO streams for I/O operations */
    BIO *in = NULL;          /* Input stream */
    BIO *b64 = NULL;         /* Base64 filter (if needed) */
    BIO *derout = NULL;      /* Output stream for DER format */
    
    /* Buffer management */
    BUF_MEM *buf = NULL;     /* Main buffer for holding data */
    
    /* Stack for strparse offsets - allows drilling into nested structures */
    STACK_OF(OPENSSL_STRING) *osk = NULL;
    
    /* Configuration and generation strings */
    char *genstr = NULL;     /* ASN.1 generation string from command line */
    char *genconf = NULL;    /* Config file for ASN.1 generation */
    
    /* File paths */
    char *infile = NULL;     /* Input file path */
    char *oidfile = NULL;    /* OID definition file path */
    char *derfile = NULL;    /* Output file path for DER data */
    
    /* Data pointers for PEM parsing */
    unsigned char *str = NULL;   /* Pointer to data buffer */
    char *name = NULL;            /* PEM label name (e.g., "CERTIFICATE") */
    char *header = NULL;          /* PEM headers */
    char *prog;                   /* Program name for error messages */
    const unsigned char *ctmpbuf; /* Const temp buffer for d2i functions */
    
    /* Control flags */
    int indent = 0;          /* Whether to indent output */
    int noout = 0;           /* Suppress output if set */
    int dump = 0;            /* Dump raw hex: 0=no, -1=all, >0=limited bytes */
    int informat = FORMAT_PEM; /* Input format (PEM/DER/Base64) */
    
    /* Offset and length controls for parsing subsections */
    int offset = 0;          /* Starting offset into data */
    int ret = 1;             /* Return code: 1=error, 0=success */
    int i, j;                /* Loop counters */
    long num;                /* Number of bytes read/processed */
    long tmplen;             /* Temporary length during strparse */
    unsigned char *tmpbuf;   /* Temporary buffer pointer during strparse */
    unsigned int length = 0; /* Length of data to parse (0 = all) */
    
    OPTION_CHOICE o;         /* Current option being processed */
    const ASN1_ITEM *it = NULL; /* ASN.1 item type for typed parsing */

    /* Initialize option parser with program name */
    prog = opt_init(argc, argv, asn1parse_options);

    /* Create stack for storing strparse offset values */
    if ((osk = sk_OPENSSL_STRING_new_null()) == NULL) {
        BIO_printf(bio_err, "%s: Memory allocation failure\n", prog);
        goto end;
    }

    /*
     * Command line option parsing loop
     * Process each option until we reach end of arguments
     */
    while ((o = opt_next()) != OPT_EOF) {
        switch (o) {
        case OPT_EOF:
        case OPT_ERR:
 opthelp:
            BIO_printf(bio_err, "%s: Use -help for summary.\n", prog);
            goto end;
        case OPT_HELP:
            opt_help(asn1parse_options);
            ret = 0;
            goto end;
        case OPT_INFORM:
            /* Parse input format string (PEM, DER, B64) */
            if (!opt_format(opt_arg(), OPT_FMT_ASN1, &informat))
                goto opthelp;
            break;
        case OPT_IN:
            infile = opt_arg();
            break;
        case OPT_OUT:
            derfile = opt_arg();
            break;
        case OPT_INDENT:
            indent = 1;
            break;
        case OPT_NOOUT:
            noout = 1;
            break;
        case OPT_OID:
            oidfile = opt_arg();
            break;
        case OPT_OFFSET:
            /* Parse starting offset - no error checking (TECHNICAL DEBT) */
            offset = strtol(opt_arg(), NULL, 0);
            break;
        case OPT_LENGTH:
            /* Parse length to process - no error checking (TECHNICAL DEBT) */
            length = strtol(opt_arg(), NULL, 0);
            break;
        case OPT_DUMP:
            /* Enable unlimited hex dump of unknown data */
            dump = -1;
            break;
        case OPT_DLIMIT:
            /* Limit hex dump to specified number of bytes */
            dump = strtol(opt_arg(), NULL, 0);
            break;
        case OPT_STRPARSE:
            /* 
             * Add offset to strparse stack
             * Multiple -strparse options allow drilling into nested structures
             * Each offset is applied sequentially to dig deeper
             */
            if (sk_OPENSSL_STRING_push(osk, opt_arg()) <= 0)
                goto end;
            break;
        case OPT_GENSTR:
            /* ASN.1 generation string (alternative to reading file) */
            genstr = opt_arg();
            break;
        case OPT_GENCONF:
            /* Configuration file for ASN.1 generation */
            genconf = opt_arg();
            break;
        case OPT_STRICTPEM:
            /* 
             * Obsolete option kept for backward compatibility
             * Modern code should use -inform pem instead
             */
            informat = FORMAT_PEM;
            break;
        case OPT_ITEM:
            /*
             * Parse as specific ASN.1 item type (e.g., X509, RSAPublicKey)
             * Provides typed parsing instead of generic ASN.1 dump
             */
            it = ASN1_ITEM_lookup(opt_arg());
            if (it == NULL) {
                size_t tmp;

                BIO_printf(bio_err, "Unknown item name %s\n", opt_arg());
                BIO_puts(bio_err, "Supported types:\n");
                /* List all available ASN.1 item types */
                for (tmp = 0;; tmp++) {
                    it = ASN1_ITEM_get(tmp);
                    if (it == NULL)
                        break;
                    BIO_printf(bio_err, "    %s\n", it->sname);
                }
                goto end;
            }
            break;
        }
    }

    /* Verify no unexpected trailing arguments */
    if (!opt_check_rest_arg(NULL))
        goto opthelp;

    /*
     * Load custom OID definitions if specified
     * Allows recognition of proprietary or extended OIDs
     */
    if (oidfile != NULL) {
        in = bio_open_default(oidfile, 'r', FORMAT_TEXT);
        if (in == NULL)
            goto end;
        OBJ_create_objects(in);
        BIO_free(in);
    }

    /* Open input file (or stdin if not specified) */
    if ((in = bio_open_default(infile, 'r', informat)) == NULL)
        goto end;

    /* Open output file for DER format if specified */
    if (derfile && (derout = bio_open_default(derfile, 'w', FORMAT_ASN1)) == NULL)
        goto end;

    /* Allocate main buffer for holding ASN.1 data */
    if ((buf = BUF_MEM_new()) == NULL)
        goto end;
    
    /*
     * Data loading section - three paths:
     * 1. Generate from config (genconf/genstr set)
     * 2. Read PEM format (default)
     * 3. Read DER or Base64 format
     */
    if (genconf == NULL && genstr == NULL && informat == FORMAT_PEM) {
        /*
         * PEM Format Path
         * Read PEM encoded data including label and headers
         * PEM format looks like:
         *   -----BEGIN CERTIFICATE-----
         *   <base64 data>
         *   -----END CERTIFICATE-----
         */
        if (PEM_read_bio(in, &name, &header, &str, &num) != 1) {
            BIO_printf(bio_err, "Error reading PEM file\n");
            ERR_print_errors(bio_err);
            goto end;
        }
        /* Point buffer directly at decoded PEM data (saves a copy) */
        buf->data = (char *)str;
        buf->length = buf->max = num;
    } else {
        /*
         * DER/Base64/Generation Path
         * Pre-allocate buffer with reasonable size (8 * BUFSIZ)
         * BUFSIZ is typically 8192, so this allocates ~64KB
         */
        if (!BUF_MEM_grow(buf, BUFSIZ * 8))
            goto end;

        if (genstr || genconf) {
            /*
             * Generation Path
             * Create ASN.1 structure from config file or string
             * Used for testing and creating sample structures
             */
            num = do_generate(genstr, genconf, buf);
            if (num < 0) {
                ERR_print_errors(bio_err);
                goto end;
            }
        } else {
            /*
             * File Reading Path (DER or Base64)
             */

            if (informat == FORMAT_BASE64) {
                /*
                 * Base64 Decoding Setup
                 * Insert base64 filter into BIO chain to decode on-the-fly
                 * Chain becomes: base64_filter -> input_file
                 */
                BIO *tmp;

                if ((b64 = BIO_new(BIO_f_base64())) == NULL)
                    goto end;
                BIO_push(b64, in);
                tmp = in;
                in = b64;
                b64 = tmp;
            }

            /*
             * Read loop - accumulate data into buffer
             * Grows buffer as needed to accommodate entire file
             */
            num = 0;
            for (;;) {
                /* Ensure buffer has room for another chunk */
                if (!BUF_MEM_grow(buf, num + BUFSIZ))
                    goto end;
                /* Read next chunk */
                i = BIO_read(in, &(buf->data[num]), BUFSIZ);
                if (i <= 0)
                    break;
                /*
                 * Overflow protection
                 * Make sure num doesn't overflow when adding i
                 * NOTE: Check happens after read but before use (TECHNICAL DEBT)
                 */
                if (i > LONG_MAX - num)
                    goto end;
                num += i;
            }
        }
        /* Point str at the data we just read/generated */
        str = (unsigned char *)buf->data;
    }

    /*
     * STRPARSE Processing Section
     * 
     * If -strparse options were given, drill down through nested ASN.1 structures
     * Each strparse value is an offset into the current structure
     * This allows examining deeply nested ASN.1 without parsing the whole tree
     * 
     * Example: Certificate -> TBSCertificate -> Extensions -> Specific Extension
     */
    if (sk_OPENSSL_STRING_num(osk)) {
        tmpbuf = str;
        tmplen = num;
        
        /* Process each strparse offset in sequence */
        for (i = 0; i < sk_OPENSSL_STRING_num(osk); i++) {
            ASN1_TYPE *atmp;
            int typ;
            
            /* Parse offset value - no error checking (TECHNICAL DEBT) */
            j = strtol(sk_OPENSSL_STRING_value(osk, i), NULL, 0);
            
            /* Validate offset is within current buffer bounds */
            if (j <= 0 || j >= tmplen) {
                BIO_printf(bio_err, "'%s' is out of range\n",
                           sk_OPENSSL_STRING_value(osk, i));
                continue;
            }
            
            /* Advance to specified offset */
            tmpbuf += j;
            tmplen -= j;
            
            /* Free previous parsed structure if any */
            atmp = at;
            ctmpbuf = tmpbuf;
            
            /* Parse ASN.1 structure at this location */
            at = d2i_ASN1_TYPE(NULL, &ctmpbuf, tmplen);
            ASN1_TYPE_free(atmp);
            
            if (!at) {
                BIO_printf(bio_err, "Error parsing structure\n");
                ERR_print_errors(bio_err);
                goto end;
            }
            
            /* Get ASN.1 type of parsed structure */
            typ = ASN1_TYPE_get(at);
            
            /*
             * Reject types that can't be recursively parsed
             * These are primitive types with no internal structure
             */
            if ((typ == V_ASN1_OBJECT)
                || (typ == V_ASN1_BOOLEAN)
                || (typ == V_ASN1_NULL)) {
                BIO_printf(bio_err, "Can't parse %s type\n", ASN1_tag2str(typ));
                ERR_print_errors(bio_err);
                goto end;
            }
            
            /*
             * This is admittedly a bit of a hack:
             * We're directly accessing the internal data pointer of the ASN1_TYPE
             * This lets us continue drilling without copying data
             * The comment in original code says "evil but it works"
             * 
             * It works because:
             * - ASN1_TYPE wraps an ASN1_STRING for constructed types
             * - The data pointer points into our original buffer
             * - We keep 'at' alive so the pointer stays valid
             */
            tmpbuf = at->value.asn1_string->data;
            tmplen = at->value.asn1_string->length;
        }
        
        /* After all strparse operations, update main pointers */
        str = tmpbuf;
        num = tmplen;
    }

    /*
     * Offset and Length Validation
     * Ensure requested offset and length are within buffer bounds
     */
    if (offset < 0 || offset >= num) {
        BIO_printf(bio_err, "Error: offset out of range\n");
        goto end;
    }

    /* Adjust for offset */
    num -= offset;

    /* If no length specified, or length exceeds remaining data, use all remaining */
    if (length == 0 || length > (unsigned int)num)
        length = (unsigned int)num;
    
    /*
     * DER Output Section
     * If output file specified, write raw DER data (no parsing/formatting)
     */
    if (derout != NULL) {
        if (BIO_write(derout, str + offset, length) != (int)length) {
            BIO_printf(bio_err, "Error writing output\n");
            ERR_print_errors(bio_err);
            goto end;
        }
    }
    
    /*
     * Display Section
     * Parse and display the ASN.1 structure (unless -noout specified)
     */
    if (!noout) {
        const unsigned char *p = str + offset;

        if (it != NULL) {
            /*
             * Typed Parsing Path
             * Parse as specific ASN.1 type (e.g., X509, RSAPublicKey)
             * Provides structured output with field names
             */
            ASN1_VALUE *value = ASN1_item_d2i(NULL, &p, length, it);
            if (value == NULL) {
                BIO_printf(bio_err, "Error parsing item %s\n", it->sname);
                ERR_print_errors(bio_err);
                goto end;
            }
            /* Pretty-print the structure with field names and values */
            ASN1_item_print(bio_out, value, 0, it, NULL);
            ASN1_item_free(value, it);
        } else {
            /*
             * Generic Parsing Path
             * Parse as generic ASN.1 and display structure
             * Shows tags, lengths, and data in hierarchical format
             * 
             * dump parameter controls hex display:
             *   0 = no hex dump
             *  -1 = unlimited hex dump
             *  >0 = dump first N bytes
             */
            if (!ASN1_parse_dump(bio_out, p, length, indent, dump)) {
                ERR_print_errors(bio_err);
                goto end;
            }
        }
    }
    
    /* Success! */
    ret = 0;
    
 end:
    /*
     * Cleanup Section
     * Free all allocated resources
     * Using goto for cleanup is an old C pattern that ensures
     * proper cleanup regardless of which error path was taken
     */
    BIO_free(derout);
    BIO_free(in);
    BIO_free(b64);
    if (ret != 0)
        ERR_print_errors(bio_err);
    BUF_MEM_free(buf);
    OPENSSL_free(name);
    OPENSSL_free(header);
    ASN1_TYPE_free(at);
    sk_OPENSSL_STRING_free(osk);
    return ret;
}

/*
 * ============================================================================
 * do_generate - Generate ASN.1 Structure from Configuration
 * ============================================================================
 *
 * PURPOSE:
 *   Creates ASN.1 encoded data from either a direct string specification
 *   or from a configuration file. This allows building complex ASN.1
 *   structures without writing manual encoding code.
 *
 * FUNCTION SIGNATURE:
 *   static int do_generate(char *genstr, const char *genconf, BUF_MEM *buf)
 *
 * ============================================================================
 * INPUTS
 * ============================================================================
 *
 * PARAMETERS:
 *   genstr  [in/optional]  ASN.1 generation string specification
 *                          Examples: "SEQUENCE:seq_section"
 *                                   "INTEGER:42"
 *                                   "UTF8:Hello World"
 *                          If NULL and genconf provided, reads from config
 *                          
 *   genconf [in/optional]  Path to configuration file
 *                          Must be in OpenSSL CONF format
 *                          If NULL, genstr must be provided
 *                          
 *   buf     [out]          Buffer to store generated DER encoded result
 *                          Must be pre-allocated (BUF_MEM_new)
 *                          Function will grow buffer as needed
 *
 * CONFIGURATION FILE FORMAT:
 *   [default]
 *   asn1 = SEQUENCE:my_sequence    # Points to the structure definition
 *   
 *   [my_sequence]
 *   field1 = INTEGER:42            # Integer field with value 42
 *   field2 = UTF8:Hello            # UTF8 string field
 *   field3 = OCTETSTRING:FORMAT:HEX,0102030405  # Hex byte string
 *   field4 = SEQUENCE:nested       # Nested sequence
 *   
 *   [nested]
 *   subfield = BOOLEAN:TRUE        # Boolean field
 *
 * SUPPORTED ASN.1 TYPES:
 *   INTEGER, BOOLEAN, NULL, OBJECT, OCTETSTRING, BITSTRING,
 *   UTF8, SEQUENCE, SET, IA5STRING, PRINTABLESTRING, T61STRING,
 *   UNIVERSALSTRING, BMPSTRING, UTCTIME, GENERALIZEDTIME
 *
 * ============================================================================
 * OUTPUTS
 * ============================================================================
 *
 * RETURN VALUE:
 *   Positive integer (>0):  Length of generated DER data (success)
 *                           The data is placed in buf->data
 *                           buf->length is NOT updated (caller's responsibility)
 *   
 *   -1:                     Error occurred
 *                           OpenSSL error stack will contain details
 *                           buf contents are undefined
 *
 * SIDE EFFECTS:
 *   - buf->data:   Contains DER encoded ASN.1 structure (on success)
 *   - buf->max:    May be increased if buffer was grown
 *   - Error stack: Populated on failure (use ERR_print_errors)
 *
 * ============================================================================
 * VARIABLE DICTIONARY
 * ============================================================================
 *
 * FUNCTION PARAMETERS:
 *   genstr           [in]      ASN.1 generation string specification
 *                              NULL = read from genconf file
 *                              
 *   genconf          [in]      Configuration file path
 *                              NULL = use genstr directly
 *                              
 *   buf              [in/out]  Buffer for DER output
 *                              IN:  Pre-allocated, possibly empty
 *                              OUT: Contains DER data, size = return value
 *
 * LOCAL VARIABLES:
 *   cnf              [local]   Configuration file object (CONF*)
 *                              Loaded from genconf file
 *                              Freed before return
 *                              
 *   atyp             [local]   Generated ASN.1 structure (ASN1_TYPE*)
 *                              Result of parsing genstr with config
 *                              Contains the complete ASN.1 tree
 *                              Freed before return
 *                              
 *   len              [local]   Length of DER encoded output (int)
 *                              First calculated, then used to grow buffer
 *                              Returned to caller on success
 *                              
 *   p                [local]   Pointer for DER encoding (unsigned char*)
 *                              Points into buf->data
 *                              Advanced by i2d_ASN1_TYPE during encoding
 *
 * ============================================================================
 * ALGORITHM FLOW
 * ============================================================================
 *
 * STEP 1: Load Configuration (if genconf provided)
 *   - Open and parse configuration file
 *   - If genstr is NULL, read "asn1" value from "default" section
 *   - Validate that genstr is now populated
 *
 * STEP 2: Generate ASN.1 Structure
 *   - Parse genstr according to configuration (or standalone)
 *   - Build internal ASN1_TYPE tree representation
 *   - Free configuration file (no longer needed)
 *   - Validate structure was created successfully
 *
 * STEP 3: Calculate Required Buffer Size
 *   - Call i2d_ASN1_TYPE with NULL pointer
 *   - Returns length needed for DER encoding
 *   - Validate length is positive
 *
 * STEP 4: Grow Buffer and Encode
 *   - Grow buf to required length
 *   - Point p at buf->data
 *   - Call i2d_ASN1_TYPE to actually encode
 *   - p is advanced past encoded data (not used again)
 *
 * STEP 5: Cleanup and Return
 *   - Free ASN1_TYPE structure
 *   - Return length of encoded data
 *   - On error: cleanup and return -1
 *
 * ============================================================================
 * MEMORY MANAGEMENT
 * ============================================================================
 *
 * ALLOCATED BY THIS FUNCTION:
 *   cnf:     Allocated by app_load_config()
 *            Freed by NCONF_free() before return (both paths)
 *            
 *   atyp:    Allocated by ASN1_generate_nconf()
 *            Freed by ASN1_TYPE_free() before return (both paths)
 *
 * MANAGED BY CALLER:
 *   buf:     Caller must allocate before call (BUF_MEM_new)
 *            Caller must free after use (BUF_MEM_free)
 *            This function only grows it, doesn't allocate or free
 *
 * ERROR PATH CLEANUP:
 *   Both cnf and atyp are NULL-safe to free
 *   The 'err' label ensures both are freed even on error
 *
 * ============================================================================
 * ERROR CONDITIONS
 * ============================================================================
 *
 * FAILURE CASES:
 *   1. Cannot load configuration file (genconf invalid)
 *      - Returns -1
 *      - Error: "Can't load config"
 *      
 *   2. No "asn1" entry in config default section
 *      - Returns -1  
 *      - Error: "Can't find 'asn1' in ''"
 *      
 *   3. ASN.1 generation fails (invalid genstr syntax)
 *      - Returns -1
 *      - Error stack populated by ASN1_generate_nconf
 *      
 *   4. DER encoding fails (internal OpenSSL error)
 *      - Returns -1
 *      - len <= 0 after i2d_ASN1_TYPE
 *      
 *   5. Buffer grow fails (out of memory)
 *      - Returns -1
 *      - BUF_MEM_grow returns 0
 *
 * ERROR HANDLING STRATEGY:
 *   - All errors goto 'err' label
 *   - 'err' label ensures proper cleanup
 *   - OpenSSL error stack preserves detailed error info
 *   - Caller should call ERR_print_errors() on -1 return
 *
 * ============================================================================
 * USAGE EXAMPLES
 * ============================================================================
 *
 * EXAMPLE 1: Simple Direct Generation
 *   BUF_MEM *buf = BUF_MEM_new();
 *   int len = do_generate("INTEGER:42", NULL, buf);
 *   if (len > 0) {
 *       // buf->data contains DER encoded INTEGER
 *       // Use the data...
 *   }
 *   BUF_MEM_free(buf);
 *
 * EXAMPLE 2: Generate from Config File
 *   BUF_MEM *buf = BUF_MEM_new();
 *   int len = do_generate(NULL, "asn1.conf", buf);
 *   if (len > 0) {
 *       // buf->data contains DER structure from config
 *   }
 *   BUF_MEM_free(buf);
 *
 * EXAMPLE 3: Complex Sequence
 *   Config file:
 *   [default]
 *   asn1 = SEQUENCE:my_seq
 *   
 *   [my_seq]
 *   version = INTEGER:1
 *   name = UTF8:TestCert
 *   
 *   Code:
 *   BUF_MEM *buf = BUF_MEM_new();
 *   int len = do_generate(NULL, "cert.conf", buf);
 *
 * ============================================================================
 * TECHNICAL NOTES
 * ============================================================================
 *
 * DER ENCODING:
 *   - DER = Distinguished Encoding Rules
 *   - Subset of BER (Basic Encoding Rules)
 *   - Produces canonical, unambiguous encoding
 *   - Used in X.509 certificates and cryptographic signatures
 *   - Format: Tag-Length-Value (TLV)
 *
 * TWO-PASS ENCODING:
 *   First pass:  i2d_ASN1_TYPE(atyp, NULL)
 *                - Calculates required buffer size
 *                - Does not allocate or write data
 *                - Returns length needed
 *   
 *   Second pass: i2d_ASN1_TYPE(atyp, &p)
 *                - Actually encodes data
 *                - Writes to buffer pointed by p
 *                - Advances p past encoded data
 *
 * WHY FREE cnf EARLY:
 *   Configuration is only needed to build the ASN1_TYPE structure
 *   Once atyp is created, cnf can be freed immediately
 *   This reduces memory footprint during encoding phase
 *
 * GENSTR SYNTAX:
 *   "TYPE:value" or "TYPE:section_name"
 *   TYPE can be primitive (INTEGER, UTF8) or constructed (SEQUENCE, SET)
 *   For constructed types, value is section name in config file
 *   For primitive types, value is the actual data
 *
 * ============================================================================
 * SECURITY CONSIDERATIONS
 * ============================================================================
 *
 * INPUT VALIDATION:
 *   - genstr and genconf are NOT validated for malicious input
 *   - Caller must ensure inputs are from trusted sources
 *   - Configuration file parsing could be exploited with crafted input
 *
 * BUFFER SAFETY:
 *   - BUF_MEM_grow() could fail on very large structures
 *   - No explicit size limit checking
 *   - Attacker could cause memory exhaustion with deeply nested structures
 *
 * RECOMMENDATIONS:
 *   - Validate genstr syntax before calling
 *   - Impose maximum size limits for production use
 *   - Only load configuration files from trusted locations
 *   - Consider adding timeout for complex ASN.1 generation
 *
 * ============================================================================
 * RELATED FUNCTIONS
 * ============================================================================
 *
 * CALLED BY:
 *   asn1parse_main() - When -genstr or -genconf options are used
 *
 * CALLS:
 *   app_load_config()       - Load and parse configuration file
 *   NCONF_get_string()      - Read value from config section
 *   ASN1_generate_nconf()   - Generate ASN.1 from string + config
 *   i2d_ASN1_TYPE()         - Encode ASN.1 structure to DER
 *   BUF_MEM_grow()          - Expand buffer to required size
 *   NCONF_free()            - Free configuration object
 *   ASN1_TYPE_free()        - Free ASN.1 structure
 *
 * SEE ALSO:
 *   ASN1_generate_v3()      - Alternative generation function
 *   d2i_ASN1_TYPE()         - Decode DER to ASN1_TYPE (inverse)
 *   apps/asn1parse.c        - This file
 *
 * ============================================================================
 * TECHNICAL DEBT
 * ============================================================================
 *
 * ISSUES:
 *   1. No size limits on generated structures
 *      - Could allocate gigabytes with malicious config
 *      - Should add MAX_ASN1_SIZE check
 *      
 *   2. Error messages could be more specific
 *      - "Can't find 'asn1'" doesn't say which section
 *      - Should include more context in errors
 *      
 *   3. cnf is freed early but error path still frees it
 *      - Works because NCONF_free is NULL-safe
 *      - But confusing to reader
 *      - Should set cnf = NULL after first free
 *
 * RECOMMENDATIONS:
 *   - Add configuration validation before generation
 *   - Implement size limits for production use
 *   - Improve error message detail
 *   - Add logging for debugging complex configs
 *
 * ============================================================================
 */
static int do_generate(char *genstr, const char *genconf, BUF_MEM *buf)
{
    CONF *cnf = NULL;           /* Configuration object */
    int len;                    /* Length of generated data */
    unsigned char *p;           /* Pointer for encoding */
    ASN1_TYPE *atyp = NULL;     /* Generated ASN.1 structure */

    if (genconf != NULL) {
        /* Load configuration file */
        if ((cnf = app_load_config(genconf)) == NULL)
            goto err;
        
        /* If no explicit genstr, read from config file */
        if (genstr == NULL)
            genstr = NCONF_get_string(cnf, "default", "asn1");
        
        if (genstr == NULL) {
            BIO_printf(bio_err, "Can't find 'asn1' in '%s'\n", genconf);
            goto err;
        }
    }

    /* Generate ASN.1 structure from string specification */
    atyp = ASN1_generate_nconf(genstr, cnf);
    NCONF_free(cnf);
    cnf = NULL;

    if (atyp == NULL)
        return -1;

    /*
     * Encode to DER format
     * First pass with NULL pointer returns required length
     */
    len = i2d_ASN1_TYPE(atyp, NULL);

    if (len <= 0)
        goto err;

    /* Grow buffer to hold encoded data */
    if (!BUF_MEM_grow(buf, len))
        goto err;

    /* Second pass actually encodes the data */
    p = (unsigned char *)buf->data;
    i2d_ASN1_TYPE(atyp, &p);

    ASN1_TYPE_free(atyp);
    return len;

 err:
    /* Cleanup on error */
    NCONF_free(cnf);
    ASN1_TYPE_free(atyp);
    return -1;
}
